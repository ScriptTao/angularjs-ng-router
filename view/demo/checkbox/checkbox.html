单选框读取值

示例：

单选1
单选2
单选3
单选4

代码：

HTML 
        
            <input ng-model="params.checkbox1" type="checkbox">单选1 

            <input ng-model="params.checkbox2" type="checkbox">单选2 

            <input ng-model="params.checkbox3" type="checkbox">单选3 

            <input ng-model="params.checkbox4" type="checkbox" readonly>单选4 

            <button ng-click="checkFn()">获取选中项</button>
            <button ng-click="check4Fn()">设置4选中</button>
JS 
        
            /**
 * Created by 脚本 on 2019/6/10.
 */
monitor.controller('checkboxCtrl', function ($scope, $state, $stateParams, ngDialog , $interval) {
    $scope.params = {
        checkbox1:true,
        checkbox2:true,
        checkbox3:false,
        checkbox4:false,
    }
    $scope.checkFn = function () {
        if($scope.params.checkbox1){
            alert('单选1')
        }
        if($scope.params.checkbox2){
            alert('单选2')
        }
        if($scope.params.checkbox3){
            alert('单选3')
        }
        if($scope.params.checkbox4){
            alert('单选4')
        }
    }
    $scope.check4Fn = function () {
        $scope.params.checkbox4 = true;
    }
})
