创建用于展示图表的html容器，其样式大小按照html规范设置

 <div id="chart" style="width: 600px;height:500px;"></div>

绑定标签数据，展示图表


$(document).ready(function(){
    makePie();
});

function makePie() {
    //
    // ustb专用配置
    // opts.ustb.xAxis 横坐标tag属性名称，如果没配置，则默认取第一个属性作为横坐标
    // opts.ustb.tag     图表数据对应的tag名称
    // opts.ustb.showLegend 是否显示图例
    //
    // 其余配置与echart配置一致，封装已经对设置进行了默认处理，只需要设置title即可
    //
    ustb.makePie("#chart", {
        "ustb": {
            "tag": "F2Curve",
            "showLegend": true
        },
        "title": {
            "text": 'F2Curve曲线',
            "subtext": '曲线数据可以点击按钮修改'
        }
    });
}

需要注意几点：

  • 和线图、柱状图不同，饼图只显示tag中的一条数据线
  • 饼图的横坐标配置和线图、柱状图相同
