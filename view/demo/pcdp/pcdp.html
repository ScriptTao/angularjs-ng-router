中间件功能展示:

GetHmiParam

获取中间件变量值 
        
            GetHmiParam.save({
                  tag_list: "Cut_Info,asd213"
            },function (res) {
                  console.log(res);
            })

GetLogData

获取中间件日志信息 
        
            var data = {
        "startTime": "",
        "endTime": "",
        "keyword": "",
        "count": 3
    };
    GetLogData.save(data, function (response) {
        console.log(response);
    });

SendTag

修改中间件变量值 
        
             var tags = [
        {
            "name": "asd213",
            "value": "test"
        },
    ];
    sendTag(tags, function (response) {
        console.log(response);
    });

SendMessage

向中间件发送消息 
        
         SendMessage.save({
        "id": "msg",
        "timeout": 3,
        "reply": true,
        "data": [
            {
                "x": "1",
                "y": "2"
            },
            {
                "x": "3",
                "y": "4"
            }
        ]
    },function (res) {
        console.log(res);
    })
