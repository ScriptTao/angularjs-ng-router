input/select读取对应值

示例


{{params.inp}}/{{params.sle}}

代码

HTML 
        
            <input type="text" ng-model="params.inp">
            <select ng-model="params.sle">
                <option value="选择1">选择1</option>
                <option value="选择2">选择2</option>
            </select>
                <b> { {params.inp } }/ { {params.sle } }</b> 
            <button ng-click="get()">获取值</button>
            <button ng-click="set()">设置值</button>
JS 
        
            monitor.controller('input_selectCtrl', function ($scope, $state, $stateParams, ngDialog , $interval) {
                $scope.params = {
                    inp:"first",
                    sle:"选择2"
                }
                $scope.get = function () {
                    alert(''+$scope.params.inp+','+$scope.params.sle+'')
                }
                $scope.set = function () {
                    $scope.params.inp="test";
                    $scope.params.sle="选择1";
                }
            })
